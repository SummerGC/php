1.创建数据库
CREATE DATABASE my_database;

2.删除数据库
DROP DATABASE my_database;

3.创建表
CREATE TABLE customers(
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(255),
    email VARCHAR(255),
    address VARCHAR(255),
    city VARCHAR(255),
    state VARCHAR(255),
    PRIMARY KEY(id)
);
